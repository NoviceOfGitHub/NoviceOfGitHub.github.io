# 构件追溯扣分情况（结束） 临时

// design
SELECT format(-(count(a.bim_id) / 50), 2) AS "扣分"
FROM (
	SELECT t1.bim_id, t1.plan_time, t2.created_date
	FROM t_trace_component_0 t1
		LEFT JOIN t_trace_component_scan_0 t2 ON t1.bim_id = t2.bim_id
	WHERE t2.project_id = #{projectId}
		AND (t2.phase_id = 10
			OR t2.phase_id = 3)
		AND t1.plan_time <= #{endDate}
		AND t1.plan_time >= #{startDate}
		AND t2.created_date IS NULL
) a

# 新增构件追溯构件数量（开始，结束）

//design
SELECT count(id) AS "数量"
FROM t_trace_component_scan_0
WHERE project_id = #{projectId}
	AND created_date >= #{startDate}
	AND created_date <= #{endDate}
	AND (phase_id = 1
		OR phase_id = 8
		OR phase_id = 15)

# 新增构件追溯模型数量（开始，结束）

//design
SELECT count(t2.id) AS "数量"
FROM t_model_version t2
	LEFT JOIN t_bim_model t1 ON t2.model_id = t1.id, (
		SELECT max(id) AS ids
		FROM t_model_version
		GROUP BY model_id
	) t3
WHERE t1.project_id = #{projectId}
	AND t3.ids = t2.id
	AND t2.created >= #{startDate}
	AND t2.created <= #{endDate}
	AND t2.stage = 2
	AND t2.status = 3

# 新增构件扫码次数

//design
SELECT ifnull(sum(出场数量), 0) AS "出场数量"
	, ifnull(sum(进场数量), 0) AS "进场数量"
	, ifnull(sum(安装数量), 0) AS "安装数量"
FROM (
	SELECT CASE phase
			WHEN '2' THEN count(id)
			ELSE 0
		END AS "出场数量"
		, CASE phase
			WHEN '3' THEN count(id)
			ELSE 0
		END AS "进场数量"
		, CASE phase
			WHEN '5' THEN count(id)
			ELSE 0
		END AS "安装数量"
	FROM (
		SELECT DISTINCT t2.id AS id, t3.trace_phase_id AS phase
		FROM t_trace_component_scan_0 t2
			LEFT JOIN t_trace_phase_type t3 ON t2.phase_id = t3.id
		WHERE t2.project_id = #{projectId}
			AND date(t2.created_date) >= #{startDate}
			AND date(t2.created_date) <= #{endDate}
	) a
	GROUP BY phase
) b

# 最新构件扫码时间

//design
SELECT ifnull(max(最新出场时间), '无') AS "最新出场时间"
	, ifnull(max(最新进场时间), '无') AS "最新进场时间"
	, ifnull(max(最新安装时间), '无') AS "最新安装时间"
FROM (
	SELECT CASE phase
			WHEN '2' THEN time
			ELSE NULL
		END AS "最新出场时间"
		, CASE phase
			WHEN '3' THEN time
			ELSE NULL
		END AS "最新进场时间"
		, CASE phase
			WHEN '5' THEN time
			ELSE NULL
		END AS "最新安装时间"
	FROM (
		SELECT t3.trace_phase_id AS phase
			, ifnull(MAX(DATE_FORMAT(t2.created_date, '%Y-%m-%d')), NULL) AS time
		FROM t_base_project t1
			LEFT JOIN t_trace_component_scan_0 t2 ON t1.id = t2.project_id
			LEFT JOIN t_trace_phase_type t3 ON t2.phase_id = t3.id
		WHERE t1.id = #{projectId}
			AND t2.created_date >= #{startDate}
			AND t2.created_date <= #{endDate}
		GROUP BY t3.trace_phase_id
	) a
) b
