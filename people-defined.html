# X月人员扣分情况

// pmp
SELECT sum(score) AS "扣分"
FROM (
	SELECT percent
		, CASE percent
			WHEN '大' THEN 0
			WHEN '小' THEN IF(count > 4, (-(count - 4)) / 2, 0)
			WHEN '无' THEN IF(count > 2, (-(count - 2)) / 2, 0)
		END AS score
	FROM (
		SELECT *
		FROM (
			SELECT dd.date AS date, dd.percent AS percent, dd.count AS count
			FROM (
				SELECT d.date, d.percent
					, @n := IF(@pre = percent, @n + 1, @n := 1) AS count
					, @pre := percent
				FROM (
					SELECT tt.date
						, ifnull(concat(CONVERT(t6.attendencenum / t3.registernum * 100, DECIMAL(10, 2)), '%'), 0) AS "出勤率"
						, ifnull(t6.attendencenum, 0) AS "出勤人数"
						, ifnull(t3.registernum, 0) AS "在册人数"
						, CASE 
							WHEN ifnull(CONVERT(t6.attendencenum / t3.registernum, DECIMAL(10, 2)), 0) >= 0.5 THEN '大'
							WHEN ifnull(CONVERT(t6.attendencenum / t3.registernum, DECIMAL(10, 2)), 0) = 0 THEN '无'
							ELSE '小'
						END AS percent
					FROM (
						SELECT @cdate := date_add(@cdate, INTERVAL -1 DAY) AS date
						FROM (
							SELECT @cdate := date_add(#{endDate}, INTERVAL +1 DAY)
							FROM t_base_project
						) d1
						WHERE @cdate > #{startDate}
					) tt
						LEFT JOIN (
							SELECT ifnull(t0.date, 0) AS date
								, ifnull(count(DISTINCT t1.id), 0) AS registernum
							FROM (
								SELECT @cdate := date_add(@cdate, INTERVAL -1 DAY) AS date
								FROM (
									SELECT @cdate := date_add(#{endDate}, INTERVAL +1 DAY)
									FROM t_base_project
								) d1
								WHERE @cdate > #{startDate}
							) t0
								LEFT JOIN t_labor_staff t1 ON 1 = 1
								LEFT JOIN t_labor_attendance t2 ON t1.id = t2.staff_id
							WHERE t1.project_id = #{projectId}
								AND t1.del = 0
								AND datediff(t0.date, t2.work_day) BETWEEN 0 AND 5
							GROUP BY t0.date
						) t3
						ON tt.date = t3.date
						LEFT JOIN (
							SELECT t0.date AS date, ifnull(count(DISTINCT t1.id), 0) AS attendencenum
							FROM (
								SELECT @cdate := date_add(@cdate, INTERVAL -1 DAY) AS date
								FROM (
									SELECT @cdate := date_add(#{endDate}, INTERVAL +1 DAY)
									FROM t_base_project
								) d1
								WHERE @cdate > #{startDate}
							) t0
								LEFT JOIN t_labor_staff t1 ON 1 = 1
								LEFT JOIN t_labor_attendance t2 ON t1.id = t2.staff_id
							WHERE t1.project_id = #{projectId}
								AND t1.del = 0
								AND t2.work_day = t0.date
							GROUP BY t0.date
						) t6
						ON t3.date = t6.date
					GROUP BY tt.date
					ORDER BY tt.date
				) d, (
						SELECT @pre := NULL, @n := 1
					) r
			) dd
		) ddd
			LEFT JOIN (
				SELECT dd.date AS date2, dd.count AS count2
				FROM (
					SELECT d.date, d.percent
						, @n := IF(@pre = percent, @n + 1, @n := 1) AS count
						, @pre := percent
					FROM (
						SELECT tt.date
							, ifnull(concat(CONVERT(t6.attendencenum / t3.registernum * 100, DECIMAL(10, 2)), '%'), 0) AS "出勤率"
							, ifnull(t6.attendencenum, 0) AS "出勤人数"

# 分公司出勤数（今天）

//pmp
SELECT count(DISTINCT staff.person_id) AS attendanceLaborCount
FROM t_base_corporation corporation
	LEFT JOIN t_base_project project ON corporation.id = project.corporation_id
	LEFT JOIN t_labor_staff staff ON project.id = staff.project_id
	LEFT JOIN t_labor_attendance attendance ON staff.id = attendance.staff_id
WHERE 1 = 1
	AND DATE_FORMAT(attendance.work_day, '%Y-%m-%d') = curdate()
	AND corporation.id = #{corporationId}

# 当日出勤率，出勤人数，在册，退场总人数，总人数，退场人数

//pmp
SELECT ifnull(concat(convert(t6.attendencenum / t3.registernum * 100, decimal(10, 2)), '%'), 0) AS percent
	, t6.attendencenum, t3.registernum, t4.allnum - t3.registernum AS walkofftotal, t4.allnum
	, count(t5.walkoffnum) AS walkoffnum
FROM (
	SELECT count(DISTINCT t1.id) AS registernum
	FROM t_labor_staff t1
		LEFT JOIN t_labor_attendance t2 ON t1.id = t2.staff_id
	WHERE t1.del = 0
		AND t1.project_id = #{projectId}
		AND datediff(curdate(), t2.work_day) <= 5
		AND datediff(curdate(), t2.work_day) >= 0
) t3
	INNER JOIN (
		SELECT count(DISTINCT t1.id) AS allnum
		FROM t_labor_staff t1
			LEFT JOIN t_labor_attendance t2 ON t1.id = t2.staff_id
		WHERE t1.del = 0
			AND t1.project_id = #{projectId}
	) t4
	INNER JOIN (
		SELECT t1.id, max(t2.work_day) AS walkoffnum
		FROM t_labor_staff t1
			LEFT JOIN t_labor_attendance t2 ON t1.id = t2.staff_id
		WHERE t1.del = 0
			AND t1.project_id = #{projectId}
		GROUP BY t1.id
		HAVING datediff(curdate(), walkoffnum) = 6
	) t5
	INNER JOIN (
		SELECT count(DISTINCT t1.id) AS attendencenum
		FROM t_labor_staff t1
			LEFT JOIN t_labor_attendance t2 ON t1.id = t2.staff_id
		WHERE t1.del = 0
			AND t1.project_id = #{projectId}
			AND t2.work_day = curdate()
	) t6

# 最近发放，分包数，时间，金额，人数（最近一次）

//pmp
SELECT count(DISTINCT t1.id) AS cornum
	, concat(t1.issue_year, '年', lpad(t1.issue_month, 2, 0), '月') AS yearmonth
	, count(t2.id) AS nop
	, ifnull(sum(t2.real_amount), 0) AS salary
FROM t_labor_salary_menu t1

WHERE t1.project_id = #{projectId}
	AND t1.status = 4
	AND t2.issue_status = 1
GROUP BY yearmonth
ORDER BY yearmonth DESC
LIMIT 1

# 新增劳务人员数不算退场的(开始，结束）

//pmp
SELECT count(DISTINCT staff.id) AS registeredNum
FROM t_labor_staff staff
	INNER JOIN t_labor_person person ON staff.person_id = person.id
	LEFT JOIN (
		SELECT id, staff_id, work_day
		FROM t_labor_attendance
	) attendance
	ON attendance.staff_id = staff.id
WHERE 1 = 1
	AND staff.del = 0
	AND staff.project_id = #{projectId}
	AND inter_date <= #{endDate}
	AND inter_date >= #{startDate}

# 平台在册总人数

//pmp
SELECT count(DISTINCT staff.id) AS registeredNum
FROM t_labor_staff staff
	INNER JOIN t_labor_person person ON staff.person_id = person.id
	LEFT JOIN (
		SELECT id, staff_id, work_day
		FROM t_labor_attendance
	) attendance
	ON attendance.staff_id = staff.id
WHERE 1 = 1
	AND staff.del = 0
	AND staff.STATUS = 2
	AND datediff(now(), attendance.work_day) <= 5
	AND staff.project_id = #{projectId}
