# 飞行图片上传次数

// pmp
SELECT ifnull(count(date_format(created, '%Y-%m-%d')), 0) AS '上传时间'
	, ifnull(count(id), 0) AS '上传次数'
FROM t_file
WHERE folder_id IN (
		SELECT id
		FROM (
			SELECT t1.id
				, if(find_in_set(parent_id, @pids) > 0, @pids := concat(@pids, ',', id), 0) AS ischild
			FROM (
				SELECT id, parent_id
				FROM t_folder t
				ORDER BY parent_id, id
			) t1, (
					SELECT @pids := (
							SELECT folder_id
							FROM t_folder_init
							WHERE name = 'fly_pictrue'
								AND project_id = #{projectId}
						) AS t4
				) t2
		) t3
		WHERE ischild != 0
	)
	AND date_format(created, '%Y-%m-%d') = curdate()

# 飞行模型上传时间

//pmp
SELECT id, name AS '时间'
FROM t_fly_model t1, (
		SELECT max(id) AS ids
		FROM t_fly_model
		GROUP BY project_id
	) t2
WHERE t1.id = t2.ids
	AND project_id = #{projectId}

# 飞行次数（开始，结束）

//pmp
SELECT count(t2.id) AS weekfly
FROM t_fly_team_info t1
	LEFT JOIN t_fly_history_data t2 ON t1.team_id = t2.team_id
WHERE FROM_UNIXTIME(start_ts / 1000, '%Y-%m-%d') > #{startDate}
	AND FROM_UNIXTIME(start_ts / 1000, '%Y-%m-%d') < #{endDate}
	AND t1.project_id = #{projectId}

# X月飞行次数及扣分情况

//pmp
SELECT count(t2.id) AS dailyfly
	, CASE 
		WHEN count(t2.id) >= 12 THEN '0'
		ELSE format((-(12 - count(t2.id))) / 10, 1)
	END AS "扣分"
FROM t_fly_team_info t1
	LEFT JOIN t_fly_history_data t2 ON t1.team_id = t2.team_id
WHERE FROM_UNIXTIME(start_ts / 1000, '%Y-%m') = #{endMonth}
	AND t1.project_id = #{projectId}

# 上一周飞行次数（结束时间之前一周）

//pmp
SELECT count(t2.id) AS flylastweek
FROM t_fly_team_info t1
	LEFT JOIN t_fly_history_data t2 ON t1.team_id = t2.team_id
WHERE dateDIFF(#{endDate}, FROM_UNIXTIME(start_ts / 1000, '%Y-%m-%d')) <= 7
	AND t1.project_id = #{projectId}

# 最近飞行时间

//pmp
SELECT ifnull(max(FROM_UNIXTIME(start_ts / 1000, '%Y-%m-%d')), '无') AS lateFlyTime
FROM t_fly_team_info info
	LEFT JOIN t_fly_history_data history ON info.team_id = history.team_id
WHERE 1 = 1
	AND FROM_UNIXTIME(start_ts / 1000, '%Y-%m-%d') >= #{startDate}
	AND FROM_UNIXTIME(start_ts / 1000, '%Y-%m-%d') <= #{endDate}
	AND info.project_id = #{projectId}
ORDER BY lateFlyTime DESC
LIMIT 1

# 飞行时长分钟（开始，结束）

//pmp
SELECT if(flyTime != 0, round(flyTime / 60, 2), 0) AS flyTotalTime
FROM (
	SELECT sum(duration) AS flyTime
	FROM t_fly_team_info info
		LEFT JOIN t_fly_history_data history ON info.team_id = history.team_id
	WHERE 1 = 1
		AND FROM_UNIXTIME(start_ts / 1000, '%Y-%m-%d') >= #{startDate}
		AND FROM_UNIXTIME(start_ts / 1000, '%Y-%m-%d') <= #{endDate}
		AND info.project_id = #{projectId}
) aa

	AND FROM_UNIXTIME(start_ts / 1000, '%Y-%m-%d') >= #{startDate}
	AND FROM_UNIXTIME(start_ts / 1000, '%Y-%m-%d') <= #{endDate}
	AND info.project_id = #{projectId}
