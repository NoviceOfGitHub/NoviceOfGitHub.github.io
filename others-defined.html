# 年月日

// pmp
SELECT year(now()) AS '年', month(now()) AS '月'
	, day(now()) AS '日'
FROM t_base_project
WHERE id = #{projectId}

# 开始，结束日期查询

//pmp
SELECT date_format(#{startDate}, '%Y年%m月%d日') AS "开始时间"
	, date_format(#{endDate}, '%Y年%m月%d日') AS "结束时间"
WHERE dd.percent = '无'
